axiom: ~~(a /\ ~a) => (a /\ ~a) end.


goal OR_1: q => q \/ p
proof
	[q:
		q \/ p
	];
	q => q \/ p
end.


goal EQ: p <=> p
proof
[p: p];
p <=> p
end.


goal DOUBLE_NEG_2: p <=> ~~p	
proof
	[p:
	[~p: F];
	~~p];
	p => ~~p;
	[~~p:
	p];
	~~p => p;
	p <=> ~~p	    
end.


goal AND_2: ~(p /\ ~p)
proof
	[p /\ ~p:
		p;
		~p;
		F
	];
	~(p /\ ~p)
end.


goal IFF_1: p <=> ~~p

		];
	~~p

		~p

goal FALSE_3: ~(p \/ q) => ~p /\ ~q
proof
	[~(p \/ q):
		[p:
			p;
			p \/ q;
			~(p \/ q);
			F
		];
		[q:
			q;
			p \/ q;
			~(p \/ q);
			F
		];
		~p;
		~q;
		~p /\ ~q
	];
	~(p \/ q) => ~p /\ ~q
end.
