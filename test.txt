axiom: ~~(a /\ ~a) => (a /\ ~a) end.


goal OR_1: q => q \/ p
proof
	[q:
		q \/ p
	];
	q => q \/ p
end.


goal EQ: p <=> p
proof
[p: p];
p <=> p
end.


goal EQB: p => p
proof
[p: p];
p <=> p
end.

end.
