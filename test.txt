axiom: ~~(a /\ ~a) => (a /\ ~a) end.


goal AND_2: ((p /\ q) => r) => (p => (q => r))
proof
	[(p /\ q) => r:
		[p: 
			[q: 
				p /\ q; 
				r
			]; 
			q => r
		]; 
		p => q => r
	];
((p /\ q) => r) => p => q => r
end.
